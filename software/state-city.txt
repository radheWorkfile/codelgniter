            public function findBranchDetails()
            {
            $post = $this->input->post();
            $result = '';
            if ($post['actnType'] == 'class_manage') {
            
            $allClassData = $this->db->select('*')->from('class_configer')->where(array('branch_id' => $post['id'] ))->get()->result_array();
            if ($allClassData != "") {
            $result .= '<option value=""> --- Select Class --- </option>';
            foreach ($allClassData as $class) {
            $result .= '<option value="' . $class['id'] . '">' . $class['class_name'] . '</option>';
            }
            } else {
            $result = '<option value="">Choose Class</option>';
            }
            }
            elseif ($post['actnType'] == 'sec_manage') {
            $classId = $post['id'];  
            if ($classId != "") {
            $allSectionData = $this->db->select('*')->from('section_configer')->where('class_id',$classId)->get()->result_array();
            if ($allSectionData != "") {
            $result .= '<option value=""> --- Select Section --- </option>';
            foreach ($allSectionData as $section) {
            $result .= '<option value="' . $section['id'] . '">' . $section['section_name'] . '</option>';
            }
            } else {
            $result = '<option value="">N/A</option>';
            }
            }
            }
            
            elseif ($post['actnType'] == 'rollNo') {
            // Array ( [id] => 2 [actnType] => rollNo [brID] => 2 [clsId] => 2 [secID] => 2 ) 
            $sectionId = $post['id'];  
            if ($sectionId != "") {
            $allRollNoData = $this->db->select('id,roll_no')->from('student')->where(array('branch_id' => $post['brID'], 'class_manage' => $post['clsId'], 'sec_manage' => $post['secID']))->order_by('id', 'DESC')->limit(1)->get()->result_array();
            if ($allRollNoData != "") {
            $lastRollNo = $allRollNoData[0]['roll_no']; 
            $nextRollNo = $lastRollNo + 1;
            $result .= $nextRollNo;
            } else {
            $result = 'N/A';
            }
            }
            }
            sleep(0.5);
            echo $result;
            }
